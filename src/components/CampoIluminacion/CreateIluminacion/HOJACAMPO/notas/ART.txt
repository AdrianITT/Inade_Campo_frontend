import React, { useEffect, useRef, useState, useMemo } from "react";
import {
  AutoComplete,
  Button,
  Card,
  Col,
  Divider,
  Form,
  Input,
  InputNumber,
  Row,
  Space,
  Switch,
  TimePicker,
  Typography,
  message,
  Collapse,
  DatePicker,
} from "antd";
import { PlusOutlined, DeleteOutlined, DownOutlined, UpOutlined } from "@ant-design/icons";
import TextArea from "antd/es/input/TextArea";
import { useBeforeUnload, useNavigationPrompt } from "../../../hooks/DetectTabClosure";

const { Text } = Typography;

const MAX_PUNTOS = 100;

const makeEmptyPunto = () => ({
  numeroPunto: null,
  areaMonitoreo: "",
  posicionTrabajo: "",
  descripcionPunto: "",
  refl_paredes: false,
  refl_plano_trabajo: false,
  lux1E2: "",
  lux2E2: "",
  lux3E2: "",
  lux1E1: "",
  lux2E1: "",
  lux3E1: "",
  nivelMinimo: null,
});

const makeEmptyBloqueART = () => ({
  fechaMonitoreo: null,
  horaInicio: null,
  horaFinal: null,
  turno: "",
  observaciones: "",
  puntos: [makeEmptyPunto()],
});

export default function FormHCArtificial({
  areasPorFila = [],
  onFinishOK,
  disabled = false,
  loading = false,
}) {
  const [form] = Form.useForm();

  // Collapses
  const [activePuntoKeys, setActivePuntoKeys] = useState({ 0: ["0"] }); // { bloqueIdx: ["puntoKey"] }

  // Dirty
  const [isDirty, setIsDirty] = useState(false);
  useBeforeUnload(isDirty);
  useNavigationPrompt(isDirty);

  const hydratingRef = useRef(true);
  useEffect(() => {
    const t = setTimeout(() => {
      hydratingRef.current = false;
    }, 0);
    return () => clearTimeout(t);
  }, []);

  const initialValues = {
    bloquesART: [makeEmptyBloqueART()],
  };

  const handleValuesChange = () => {
    if (hydratingRef.current) return;
    setIsDirty(true);
  };

  const autoCompleteOptions = useMemo(() => {
    return (Array.isArray(areasPorFila) ? areasPorFila : [])
      .filter(Boolean)
      .map((a) => ({ value: String(a) }));
  }, [areasPorFila]);

  const onFinish = (values) => {
    const bloques = Array.isArray(values?.bloquesART) ? values.bloquesART : [];

    for (let i = 0; i < bloques.length; i++) {
      const b = bloques[i] || {};
      if (!b?.fechaMonitoreo) {
        message.warning(`Bloque #${i + 1}: Falta Fecha de Monitoreo.`);
        return;
      }
      if (!b?.horaInicio || !b?.horaFinal) {
        message.warning(`Bloque #${i + 1}: Falta Hora inicio o Hora final.`);
        return;
      }
      if (!String(b?.turno || "").trim()) {
        message.warning(`Bloque #${i + 1}: Falta Turno.`);
        return;
      }
    }

    const payload = {
      tipo: "ART",
      bloques: bloques.map((b) => ({
        fechaMonitoreo: b?.fechaMonitoreo ? b.fechaMonitoreo.format("YYYY-MM-DD") : null,
        horaInicio: b?.horaInicio ? b.horaInicio.format("HH:mm") : null,
        horaFinal: b?.horaFinal ? b.horaFinal.format("HH:mm") : null,
        turno: b?.turno ?? "",
        observaciones: b?.observaciones ?? "",
        puntos: Array.isArray(b?.puntos) ? b.puntos : [],
      })),
    };

    onFinishOK?.(payload);
    message.success("Formulario ART listo (multi-bloques).");
    setIsDirty(false);
  };

  return (
    <div style={ui.page}>
      <Form
        form={form}
        layout="vertical"
        initialValues={initialValues}
        onFinish={onFinish}
        disabled={disabled}
        onValuesChange={handleValuesChange}
      >
        {/* Header sticky */}
        <div style={ui.header}>
          <div>
            <div style={ui.title}>Hoja de Campo</div>
            <div style={ui.subtitle}>Luz Artificial (ART) — Multi-bloques</div>
          </div>

          <Space wrap>
            <span style={styles.chip}>
              {isDirty ? "Cambios sin guardar" : "Sin cambios"}
            </span>

            <Button
              type="primary"
              size="large"
              htmlType="submit"
              loading={loading}
              disabled={disabled || loading || !isDirty}
            >
              Guardar cambios
            </Button>
          </Space>
        </div>

        <Divider style={ui.divider} />

        {/* LISTA DE BLOQUES ART */}
        <Form.List name="bloquesART">
          {(bloqueFields, { add: addBloque, remove: removeBloque }) => (
            <>
              <div style={ui.toolbar}>
                <Space wrap>
                  <Button
                    type="primary"
                    ghost
                    size="large"
                    icon={<PlusOutlined />}
                    onClick={() => {
                      addBloque(makeEmptyBloqueART());
                      setTimeout(() => {
                        const last = bloqueFields.length; // nuevo índice
                        setActivePuntoKeys((prev) => ({ ...prev, [last]: ["0"] }));
                      }, 0);
                    }}
                  >
                    Agregar bloque ART
                  </Button>

                  <span style={styles.chip}>Total bloques: {bloqueFields.length}</span>
                </Space>

                <Text type="secondary">Captura por bloques y puntos</Text>
              </div>

              <div style={{ display: "grid", gap: 12 }}>
                {bloqueFields.map((bloqueField, bloqueIdx) => {
                  return (
                    <Card
                      key={bloqueField.key}
                      style={styles.card}
                      bodyStyle={{ padding: 16 }}
                      title={
                        <Space style={{ width: "100%", justifyContent: "space-between" }}>
                          <b>Bloque ART #{bloqueIdx + 1}</b>
                          <Button
                            danger
                            size="middle"
                            icon={<DeleteOutlined />}
                            disabled={bloqueFields.length === 1}
                            onClick={() => {
                              removeBloque(bloqueField.name);
                              setActivePuntoKeys((prev) => {
                                const next = { ...prev };
                                delete next[bloqueIdx];
                                return next;
                              });
                            }}
                          >
                            Quitar bloque
                          </Button>
                        </Space>
                      }
                    >
                      {/* Header fields del bloque */}
                      <Row gutter={[16, 8]}>
                        <Col xs={24} sm={12} lg={8}>
                          <Form.Item
                            label="Fecha de Monitoreo"
                            name={[bloqueField.name, "fechaMonitoreo"]}
                            rules={[{ required: true, message: "Requerido" }]}
                          >
                            <DatePicker size="large" style={{ width: "100%" }} />
                          </Form.Item>
                        </Col>

                        <Col xs={24} sm={12} lg={8}>
                          <Form.Item
                            label="Hora inicio"
                            name={[bloqueField.name, "horaInicio"]}
                            rules={[{ required: true, message: "Requerido" }]}
                          >
                            <TimePicker size="large" format="HH:mm" style={{ width: "100%" }} />
                          </Form.Item>
                        </Col>

                        <Col xs={24} sm={12} lg={8}>
                          <Form.Item
                            label="Hora final"
                            name={[bloqueField.name, "horaFinal"]}
                            rules={[{ required: true, message: "Requerido" }]}
                          >
                            <TimePicker size="large" format="HH:mm" style={{ width: "100%" }} />
                          </Form.Item>
                        </Col>

                        <Col xs={24} sm={12} lg={8}>
                          <Form.Item
                            label="Turno"
                            name={[bloqueField.name, "turno"]}
                            rules={[{ required: true, message: "Requerido" }]}
                          >
                            <Input
                              size="large"
                              placeholder="Ej: Matutino / Vespertino / Nocturno"
                            />
                          </Form.Item>
                        </Col>

                        <Col xs={24} sm={12} lg={16}>
                          <Form.Item label="Observaciones" name={[bloqueField.name, "observaciones"]}>
                            <TextArea
                              placeholder="Ej: Observaciones del bloque..."
                              autoSize={{ minRows: 2, maxRows: 4 }}
                            />
                          </Form.Item>
                        </Col>
                      </Row>

                      <Divider style={ui.dividerSoft} />

                      <div style={ui.sectionRow}>
                        <div style={ui.sectionTitle}>Puntos de muestreo</div>
                        <span style={styles.chip}>
                          Total:{" "}
                          {Array.isArray(form.getFieldValue(["bloquesART", bloqueField.name, "puntos"]))
                            ? form.getFieldValue(["bloquesART", bloqueField.name, "puntos"]).length
                            : 0}
                        </span>
                      </div>

                      <div style={{ height: 10 }} />

                      {/* PUNTOS DENTRO DE CADA BLOQUE */}
                      <Form.List name={[bloqueField.name, "puntos"]}>
                        {(fields, { add, remove }) => (
                          <>
                            <Space style={{ marginBottom: 12 }} wrap>
                              <Button
                                type="default"
                                size="large"
                                icon={<PlusOutlined />}
                                onClick={() => {
                                  if (fields.length >= MAX_PUNTOS) {
                                    return message.warning(
                                      `Máximo ${MAX_PUNTOS} puntos por bloque.`
                                    );
                                  }
                                  add(makeEmptyPunto());
                                  setTimeout(() => {
                                    const last = fields.length;
                                    setActivePuntoKeys((prev) => ({
                                      ...prev,
                                      [bloqueIdx]: [String(last)],
                                    }));
                                  }, 0);
                                }}
                                disabled={fields.length >= MAX_PUNTOS}
                              >
                                Agregar punto
                              </Button>

                              <span style={styles.chip}>Total: {fields.length}</span>
                            </Space>

                            <div style={{ display: "grid", gap: 12 }}>
                              {fields.map((field, idx) => {
                                const panelKey = String(field.name);
                                const activeKeys = activePuntoKeys[bloqueIdx] ?? ["0"];

                                return (
                                  <div key={field.key} style={styles.collapseWrap}>
                                    <Collapse
                                      activeKey={activeKeys}
                                      onChange={(keys) =>
                                        setActivePuntoKeys((prev) => ({
                                          ...prev,
                                          [bloqueIdx]: Array.isArray(keys) ? keys : [keys],
                                        }))
                                      }
                                      accordion
                                      style={{ background: "transparent" }}
                                      bordered={false}
                                      expandIconPosition="end"
                                      expandIcon={({ isActive }) =>
                                        isActive ? <UpOutlined /> : <DownOutlined />
                                      }
                                      items={[
                                        {
                                          key: panelKey,
                                          label: (
                                            <Space
                                              style={{ width: "100%", justifyContent: "space-between" }}
                                            >
                                              <b>Punto #{idx + 1}</b>

                                              <Button
                                                danger
                                                size="middle"
                                                icon={<DeleteOutlined />}
                                                onClick={(e) => {
                                                  e.stopPropagation();
                                                  remove(field.name);
                                                }}
                                                disabled={fields.length === 1}
                                              >
                                                Quitar
                                              </Button>
                                            </Space>
                                          ),
                                          children: (
                                            <Card
                                              size="small"
                                              style={styles.cardInner}
                                              bodyStyle={{ padding: 14 }}
                                            >
                                              <Row gutter={[16, 8]}>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item
                                                    label="Número de punto"
                                                    name={[field.name, "numeroPunto"]}
                                                  >
                                                    <InputNumber
                                                      size="large"
                                                      style={{ width: "100%" }}
                                                      min={1}
                                                    />
                                                  </Form.Item>
                                                </Col>

                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item
                                                    label="Área monitoreada"
                                                    name={[field.name, "areaMonitoreo"]}
                                                  >
                                                    <AutoComplete
                                                      options={autoCompleteOptions}
                                                      placeholder="Ej: Cajas"
                                                    >
                                                      <Input size="large" />
                                                    </AutoComplete>
                                                  </Form.Item>
                                                </Col>

                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item
                                                    label="Posición de trabajo"
                                                    name={[field.name, "posicionTrabajo"]}
                                                  >
                                                    <Input size="large" placeholder="Ej: Operador" />
                                                  </Form.Item>
                                                </Col>
                                              </Row>

                                              <Form.Item
                                                label="Descripción de la operación"
                                                name={[field.name, "descripcionPunto"]}
                                              >
                                                <Input
                                                  size="large"
                                                  placeholder="Ej: Inspección visual / Lectura etiquetas"
                                                />
                                              </Form.Item>

                                              {/* Switches mutuamente excluyentes */}
                                              <Form.Item
                                                noStyle
                                                shouldUpdate={(prev, cur) => {
                                                  const p =
                                                    prev?.bloquesART?.[bloqueField.name]?.puntos?.[field.name] ||
                                                    {};
                                                  const c =
                                                    cur?.bloquesART?.[bloqueField.name]?.puntos?.[field.name] ||
                                                    {};
                                                  return (
                                                    p.refl_paredes !== c.refl_paredes ||
                                                    p.refl_plano_trabajo !== c.refl_plano_trabajo
                                                  );
                                                }}
                                              >
                                                {({ getFieldValue }) => {
                                                  const basePath = [
                                                    "bloquesART",
                                                    bloqueField.name,
                                                    "puntos",
                                                    field.name,
                                                  ];

                                                  const paredes = !!getFieldValue([
                                                    ...basePath,
                                                    "refl_paredes",
                                                  ]);
                                                  const plano = !!getFieldValue([
                                                    ...basePath,
                                                    "refl_plano_trabajo",
                                                  ]);

                                                  return (
                                                    <Row gutter={[16, 8]}>
                                                      <Col xs={24} sm={12}>
                                                        <Form.Item
                                                          label="Pared (reflexión)"
                                                          name={[field.name, "refl_paredes"]}
                                                          valuePropName="checked"
                                                        >
                                                          <Switch
                                                            disabled={plano && !paredes}
                                                            onChange={(checked) => {
                                                              if (checked) {
                                                                form.setFields([
                                                                  {
                                                                    name: [
                                                                      ...basePath,
                                                                      "refl_plano_trabajo",
                                                                    ],
                                                                    value: false,
                                                                  },
                                                                ]);
                                                              }
                                                            }}
                                                          />
                                                        </Form.Item>
                                                      </Col>

                                                      <Col xs={24} sm={12}>
                                                        <Form.Item
                                                          label="Plano de trabajo (reflexión)"
                                                          name={[field.name, "refl_plano_trabajo"]}
                                                          valuePropName="checked"
                                                        >
                                                          <Switch
                                                            disabled={paredes && !plano}
                                                            onChange={(checked) => {
                                                              if (checked) {
                                                                form.setFields([
                                                                  {
                                                                    name: [...basePath, "refl_paredes"],
                                                                    value: false,
                                                                  },
                                                                ]);
                                                              }
                                                            }}
                                                          />
                                                        </Form.Item>
                                                      </Col>
                                                    </Row>
                                                  );
                                                }}
                                              </Form.Item>

                                              <Divider style={ui.dividerSoft} />

                                              <Text strong>Lux E2 Iluminación</Text>
                                              <Row gutter={[16, 8]} style={{ marginTop: 8 }}>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="1 Lux E2" name={[field.name, "lux1E2"]}>
                                                    <Input size="large" placeholder="Ej: 350" />
                                                  </Form.Item>
                                                </Col>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="2 Lux E2" name={[field.name, "lux2E2"]}>
                                                    <Input size="large" placeholder="Ej: 360" />
                                                  </Form.Item>
                                                </Col>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="3 Lux E2" name={[field.name, "lux3E2"]}>
                                                    <Input size="large" placeholder="Ej: 340" />
                                                  </Form.Item>
                                                </Col>
                                              </Row>

                                              <Text strong>Lux E1 Reflexión</Text>
                                              <Row gutter={[16, 8]} style={{ marginTop: 8 }}>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="1 Lux E1" name={[field.name, "lux1E1"]}>
                                                    <Input size="large" placeholder="Ej: 300" />
                                                  </Form.Item>
                                                </Col>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="2 Lux E1" name={[field.name, "lux2E1"]}>
                                                    <Input size="large" placeholder="Ej: 310" />
                                                  </Form.Item>
                                                </Col>
                                                <Col xs={24} sm={12} lg={8}>
                                                  <Form.Item label="3 Lux E1" name={[field.name, "lux3E1"]}>
                                                    <Input size="large" placeholder="Ej: 295" />
                                                  </Form.Item>
                                                </Col>
                                              </Row>

                                              <Divider style={ui.dividerSoft} />

                                              <Form.Item
                                                label="Nivel mínimo de iluminación"
                                                name={[field.name, "nivelMinimo"]}
                                              >
                                                <InputNumber
                                                  size="large"
                                                  style={{ width: "100%" }}
                                                  min={0}
                                                  step={0.01}
                                                />
                                              </Form.Item>
                                            </Card>
                                          ),
                                        },
                                      ]}
                                    />
                                  </div>
                                );
                              })}
                            </div>
                          </>
                        )}
                      </Form.List>
                    </Card>
                  );
                })}
              </div>

              {/* Footer sticky (tablet) */}
              <div style={ui.footer}>
                <Button
                  type="primary"
                  size="large"
                  block
                  htmlType="submit"
                  loading={loading}
                  disabled={disabled || loading || !isDirty}
                >
                  Guardar cambios
                </Button>
              </div>
            </>
          )}
        </Form.List>
      </Form>
    </div>
  );
}

const ui = {
  page: {
    padding: 16,
    background: "#f6f8fb",
    minHeight: "100%",
  },

  header: {
    position: "sticky",
    top: 0,
    zIndex: 20,
    background: "rgba(246, 248, 251, 0.92)",
    backdropFilter: "blur(8px)",
    border: "1px solid #e6eaf2",
    borderRadius: 16,
    padding: 14,
    display: "flex",
    alignItems: "center",
    justifyContent: "space-between",
    gap: 12,
    marginBottom: 12,
  },

  title: {
    fontSize: 18,
    fontWeight: 800,
    color: "#0f172a",
    lineHeight: 1.1,
  },

  subtitle: {
    fontSize: 13,
    color: "#64748b",
    marginTop: 2,
  },

  toolbar: {
    display: "flex",
    alignItems: "center",
    justifyContent: "space-between",
    gap: 12,
    flexWrap: "wrap",
    marginBottom: 12,
  },

  sectionRow: {
    display: "flex",
    alignItems: "center",
    justifyContent: "space-between",
    gap: 10,
    flexWrap: "wrap",
  },

  sectionTitle: {
    fontWeight: 800,
    color: "#0f172a",
  },

  divider: {
    margin: "12px 0",
    borderColor: "#e6eaf2",
  },

  dividerSoft: {
    margin: "10px 0",
    borderColor: "#e6eaf2",
  },

  footer: {
    position: "sticky",
    bottom: 12,
    zIndex: 20,
    marginTop: 12,
  },
};

const styles = {
  card: {
    borderRadius: 16,
    border: "1px solid #e6eaf2",
    boxShadow: "0 8px 24px rgba(15, 23, 42, 0.05)",
    background: "#ffffff",
  },

  cardInner: {
    background: "#ffffff",
    borderRadius: 14,
  },

  chip: {
    fontSize: 12,
    color: "#475569",
    background: "#f1f5f9",
    padding: "4px 10px",
    borderRadius: 999,
    border: "1px solid #e2e8f0",
  },

  collapseWrap: {
    borderRadius: 14,
    border: "1px solid #e6eaf2",
    background: "#ffffff",
    boxShadow: "0 6px 18px rgba(15, 23, 42, 0.04)",
    overflow: "hidden",
  },
};
